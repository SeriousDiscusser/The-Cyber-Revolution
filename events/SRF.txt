#Discover Galapagos #2285
country_event = {
	id = 55008
	title = "EVTNAME55008"
	desc = "EVTDESC55008"
	picture = "galapagos"

	fire_only_once = yes

	trigger = {
		OR = {
			has_country_modifier = low_surfing_budget
			has_country_modifier = medium_surfing_budget
			has_country_modifier = high_surfing_budget
		}
	}
	mean_time_to_happen = {
		months = 60
	}
	option = {
		name = "EVTOPTA55008"
		2285 = {
			any_country = {
				limit = {owns = THIS}
				country_event = {id=55011}
			}
			secede_province = THIS
			SRF = {
				limit = {has_country_modifier = low_surfing_budget}
				set_country_flag = done_discovering
			}
		}
	}
}

#Ride the Big One
country_event = {
	id = 55010
	title = "EVTNAME55010"
	desc = "EVTDESC55010"
	picture = "wave"
	
	major = yes
	fire_only_once = yes
	
	trigger = {
		has_country_flag = done_discovering
		OR = {
			has_country_modifier = low_surfing_budget
			has_country_modifier = medium_surfing_budget
			has_country_modifier = high_surfing_budget
		}
	}
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = .5
			has_country_modifier = medium_surfing_budget
		}
		modifier = {
			factor = .25
			has_country_modifier = high_surfing_budget
		}
	}
	option = {
		name = "EVTOPTA55009"
		prestige = 100
		any_pop = {
			MIL = -5
			CON = -5
		}
		remove_country_modifier = low_surfing_budget
		remove_country_modifier = medium_surfing_budget
		remove_country_modifier = high_surfing_budget
	}
}
